#include <stdio.h>

#define ROWS 20
#define COLS 40

char canvas[ROWS][COLS];
🔹 STEP 3: Initialize the canvas
Fill everything with _

void initCanvas() {
    for(int i = 0; i < ROWS; i++) {
        for(int j = 0; j < COLS; j++) {
            canvas[i][j] = '_';
        }
    }
}
🔹 STEP 4: Display the canvas
void displayCanvas() {
    for(int i = 0; i < ROWS; i++) {
        for(int j = 0; j < COLS; j++) {
            printf("%c ", canvas[i][j]);
        }
        printf("\n");
    }
}

🔹 STEP 5: Draw a Line
Simplest version (horizontal line):

void drawLine(int row, int col, int length) {
    for(int i = col; i < col + length; i++) {
        canvas[row][i] = '*';
    }
}
🔹 STEP 6: Draw a Rectangle
void drawRectangle(int x, int y, int width, int height) {
    for(int i = y; i < y + height; i++) {
        for(int j = x; j < x + width; j++) {
            canvas[i][j] = '*';
        }
    }
}
🔹 STEP 7: Draw a Triangle (simple version)
void drawTriangle(int x, int y, int height) {
    for(int i = 0; i < height; i++) {
        for(int j = 0; j <= i; j++) {
            canvas[y + i][x + j] = '*';
        }
    }
}
🔹 STEP 8: Draw a Circle (basic logic)
Use distance formula:

void drawCircle(int centerX, int centerY, int radius) {
    for(int i = 0; i < ROWS; i++) {
        for(int j = 0; j < COLS; j++) {
            int dx = i - centerY;
            int dy = j - centerX;
            if(dx*dx + dy*dy <= radius*radius) {
                canvas[i][j] = '*';
            }
        }
    }
}
🔹 STEP 9: Delete (erase part)
Replace * with _

void eraseArea(int x, int y, int width, int height) {
    for(int i = y; i < y + height; i++) {
        for(int j = x; j < x + width; j++) {
            canvas[i][j] = '_';
        }
    }
}
